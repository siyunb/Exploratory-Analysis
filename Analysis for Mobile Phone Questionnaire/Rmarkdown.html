qingli <- function(x) {
    a = 1
    m <- vector(mode = "numeric", length = 0)
    for (i in 1:dim(x)[1]) {
        for (j in 1:dim(x)[2]) {
            if (x[i, j] > 7) {
                m[a] <- i
                a <- a + 1
            }
        }
    }
    return(m)
